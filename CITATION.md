```bibtex
  title={Classicap: A Multimodal Dataset for Classical Music Understanding with Audio Captions and Theoretical Analysis},
  author={Cui, Junhan},
  year={2025},
  month={10},
  publisher={Zenodo},
  version={1.0},
  doi={10.5281/zenodo.17394078},
  url={https://github.com/jcjcjuju/Classicap-Dataset}
}
```
